103    def create_network_df(self, run_number: int) -> pd.DataFrame:
104        """Create an empty pandas dataframe ready to populate with network analysis data
105
106        **Returns:**  
107            Pandas dataframe consisting of all possible combination of healthcare system 'nodes'. There are 6 in total. 
108            Column names of dataframes are: source, target, run_number, weight  
109
110        """
111        seq05 = np.arange(0, 6, 1, dtype=int) # Numbers from 0 to 5, not 6!
112        all_combinations = list(product(seq05, seq05))
113        index_iuc_combs = list([(6, 0), (6, 1), (6, 2), (6, 3), (6, 4), (6, 5)])
114        df = pd.DataFrame(np.concatenate((all_combinations, index_iuc_combs)), columns=['source','target'])
115        df['run_number'] = run_number
116        df['weight'] = 0
117
118        return df

122    def date_time_of_call(self, start_dt: str, elapsed_time: int) -> list[int, int, str, int, pd.Timestamp]:
123        """
124        Calculate a range of time-based parameters given a specific date-time
125
126        **Returns:**  
127            list(  
128                `dow`             : int  
129                `current_hour`    : int  
130                `weekday`         : str   
131                `current_quarter` : int  
132                `current_dt`      : datetime  
133            )
134
135        """
136        # Elapsed_time = time in minutes since simulation started
137
138        start_dt = pd.to_datetime(start_dt)
139
140        current_dt = start_dt + pd.Timedelta(elapsed_time, unit='min')
141
142        dow = current_dt.strftime('%a')
143        # 0 = Monday, 6 = Sunday
144        weekday = "weekday" if current_dt.dayofweek < 5 else "weekend"
145
146        current_hour = current_dt.hour
147
148        current_quarter = current_dt.quarter
149        
150        return [dow, current_hour, weekday, current_quarter, current_dt]

152    def generate_111_calls(self):
153        """
154        **Patient generator**
155        
156        Keeps creating new patients until current time equals sim_duration + warm_up_duration
157        
158        """
159        
160        if self.env.now < self.sim_duration + self.warm_up_duration :
161            while True:
162                self.patient_counter += 1
163                
164                # Create a new caller/patient
165                pt = Caller(self.patient_counter, self.what_if_sim_run)
166                
167                # Allocate caller to a GP surgery
168                pt.gp = random.choices(self.GP_surgery_df["gp_surgery_id"], weights=self.GP_surgery_df["prob"])[0]
169                
170                # Set caller/patient off on their healthcare journey
171                self.env.process(self.patient_journey(pt))
172                
173                # Get current day of week and hour of day
174                [dow, hod, weekday, qtr, current_dt] = self.date_time_of_call(self.start_dt, self.env.now)
175
176                # Update patient instance with time-based values so the current time is known
177                pt.day = dow
178                pt.hour = hod 
179                pt.weekday = weekday
180                pt.qtr = qtr
181
182                # This variable is in use, do not comment out!
183                weekday_bool = 1 if weekday == 'weekday' else 0
184                
185                # The interarrival times are recorded in a pandas dataframe.
186
187                # TODO - See if the interarrival times can be improved with machine learning
188
189                # Retrieve mean interarrival time for the specific hour, weekday/weekend and yearly quarter
190                inter_time = float(self.call_interarrival_times_lu.query("exit_hour == @hod & weekday == @weekday_bool & qtr == @qtr")["mean_inter_arrival_time"].tolist()[0])
191
192                # Retrieve the maximum interarrival rate for the specific hour, weekday/weekend and yearly quarter
193                max_hourly_interarrival_rate = (60 / (float(self.call_interarrival_times_lu.query("exit_hour == @hod & weekday == @weekday_bool & qtr == @qtr")["max_arrival_rate"].tolist()[0])))
194
195                # Retrieve the minimum interarrival rate for the specific hour, weekday/weekend and yearly quarter
196                min_hourly_interarrival_rate = (60 / (float(self.call_interarrival_times_lu.query("exit_hour == @hod & weekday == @weekday_bool & qtr == @qtr")["min_arrival_rate"].tolist()[0]))) - 1
197
198                # Determine the interarrival time for the next patient by sampling from the exponential distrubution
199                sampled_interarrival = random.expovariate(1.0 / inter_time) 
200
201                # There are a number of user-configurable options to try and improve inter-arrival rate estimation
202                # SPOILER: base is the best!
203
204                if self.ia_type == 'base':
205                    # Use sampled interarrival time with a check to ensure it does not go over 60 minutes
206                    # as this would technically be in the 'next' hour
207                    sampled_interarrival = 59 if sampled_interarrival >= 60 else sampled_interarrival
208                elif self.ia_type == 'max':
209                    # Repeatedly sample until a value that is less than the minimum hourly interarrival rate
210                    # is sampled. In tests, this lead to a large drop in overall patient numbers
211                    while True if sampled_interarrival <= min_hourly_interarrival_rate else False:
212                        sampled_interarrival = random.expovariate(1.0 / inter_time) 
213                elif self.ia_type == 'minmax':
214                    # Repeatedly sample until a value that is less than the minimum hourly interarrival rate
215                    # but greater than the maximum hourly interarrival rate is sampled.
216                    # In tests, this also caused a large drop in overall patient numbers
217                    while True if max_hourly_interarrival_rate >= sampled_interarrival <= min_hourly_interarrival_rate else False:
218                        sampled_interarrival = random.expovariate(1.0 / inter_time) 
219                elif self.ia_type == 'mean':
220                    # No random sampling at all. Just use the mean interarrival time
221                    # Also resulted in lower than expected patient arrivals compared to real life and the 
222                    # exponential distribuation sampling. Turns out Dan was right about the exponential distrubtion...
223                    sampled_interarrival = inter_time
224
225                # Freeze function until interarrival time has elapsed
226                yield self.env.timeout(sampled_interarrival)

228    def ooh(self, patient: Caller) -> str:
229        """
230        Determine whether the patient's index 111 call is in-hours or out-of-hours (ooh)
231        
232        """
233        if patient.weekday != 'weekday':
234            return 'ooh'
235        elif patient.hour < 8:
236            return 'ooh'
237        elif patient.hour > 18:
238            return 'ooh'
239        else:
240            return 'in-hours'

242    def patient_journey(self, patient: Caller) -> None:
243        """
244            Determine whether the patient's index 111 call is in-hours or out-of-hours (ooh)
245        """
246
247        # Variable to keep track of the order of healthcare interactions.
248        # Index_IUC is always 0
249        instance_id = 0
250
251        # Capture current simulation time
252        patient_enters_sim = self.env.now
253    
254        # Loop will keep iterating while patient time in sim is less than simulation run time + warm up duration
255        while patient.timer < (self.warm_up_duration + G.pt_time_in_sim):
256            
257            # Increment instance_id
258            instance_id += 1
259
260            # Capture current patient actvity
261            current_step = patient.activity
262
263            # Determine the patient's next healthcare interaction
264            if self.transition_type == 'tree':
265                # Determine next step using a decision tree modle
266                # SPOILER: it was terrible!
267
268                # Check whether current time is in-hours or out-of-hours
269                ooh_check = self.ooh(patient)
270
271                # Use the next_destination_tree function to run the patient through the decision tree model
272                next_step = self.t.next_destination_tree(current_step, patient.pc_outcome, patient.gp_timely_callback, patient.qtr, patient.last_step, ooh_check)
273
274                # Before the current step becomes the 'next step', capture current step as previous step
275                # in the next iteration....if that makes sense.
276                patient.last_step = current_step
277            else:
278                # Default behaviour
279                # Next destination is determined by next_destination function.
280                next_step = "GP" if ((self.what_if_sim_run == 'Yes') & (current_step == 'Index_IUC')) else self.t.next_destination(patient.activity, patient.gp_timely_callback, patient.qtr)
281
282            # If next step is ED, we need to determine whether this will be an avoidable admission or not
283            ED_urgent = "not applicable"
284
285            # Prepare source and target variables for network dataframe
286            source = next((v for k, v in self.node_lookup.items() if k == current_step), None)
287            target = next((v for k, v in self.node_lookup.items() if k == next_step), None)
288
289            # Update network graph table
290            self.network_df['weight'] = self.network_df.apply(lambda x: x['weight'] + 1 if x['source'] == source and x['target'] == target else x['weight'], axis = 1)
291
292
293            if patient.activity == 'Index_IUC':
294                # As a one-off, capture the completed Index IUC call
295                self.add_patient_result_row('completed', patient, instance_id, ED_urgent)
296            
297            # Update current patient activity
298            patient.activity = next_step
299                            
300            # Prepare data for event being 'scheduled'
301            # The time between this and 'start' will be the queue time
302            # or wait time or time for the patient to have another
303            # interaction with a healthcare service.
304            self.add_patient_result_row('scheduled', patient, instance_id, ED_urgent)
305                
306            # Consult the wait_time function to find out how long until the next activity
307            wait_time = self.t.wait_time(current_step, patient.activity)
308            yield self.env.timeout(wait_time)
309
310            # Based on current patient activity, need to determine where they go next...
311            if(patient.activity == 'End'):
312                break
313            elif patient.activity == 'ED':
314                ED_urgent = "non_urgent" if self.t.avoidable_ed_admission(patient.pc_outcome, patient.gp_timely_callback) else "urgent"
315                self.add_patient_result_row('start', patient, instance_id, ED_urgent)
316                yield self.env.process(self.step_visit(patient, instance_id, ED_urgent))
317            else:
318                self.add_patient_result_row('start', patient, instance_id, ED_urgent)
319                yield self.env.process(self.step_visit(patient, instance_id, ED_urgent))
320            
321            # Keep track of how long patient has been in sim
322            patient.timer = self.env.now - patient_enters_sim
